💻 AFTER INSTALLATION:
══════════════════════
#include <input.h>
#include <stdio.h>

int main() {
    char name[50];
    input("Name: ", name, sizeof(name));
    int age = (int)inputLong("Age: ");
    printf("Hello %s, %d!\n", name, age);
    return 0;
}

Compile: gcc myprogram.c -linput
